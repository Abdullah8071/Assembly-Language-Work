TITLE FACTORIAL PROCEDURE

INCLUDE IRVINE32.INC

.DATA
S1 BYTE "FACTORIAL : ",0

.CODE
	FACTORIAL PROC
		MOV ECX,EAX
		DEC ECX
		MOV EBX,EAX
		MOV EAX,1
		L1:
			MUL EBX
			DEC EBX
		LOOP L1
		CALL WRITEDEC
		CALL CRLF
		RET
	FACTORIAL ENDP

	START PROC
		MOV EAX,0
		MOV EBX,0
		CALL READINT
		CALL DUMPREGS
		CALL CRLF
		MOV EDX,OFFSET S1
		CALL WRITESTRING
		CALL FACTORIAL
	START ENDP

	EXIT
END START
