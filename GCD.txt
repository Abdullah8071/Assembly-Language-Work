TITLE GCD OF TWO NUMBERS

INCLUDE IRVINE32.INC

.DATA
	NUM1 DWORD ?
	NUM2 DWORD ?
	S BYTE "GCD : ",0

.CODE
	GCD PROC
		MOV EAX,NUM1
		MOV EBX,NUM2

	L1:	CMP EAX,EBX
		JA N1
		JB N2
		JE N3
		N1 : SUB EAX,EBX
		     JMP L1
		N2 : SUB EBX,EAX
			 JMP L1

	N3: RET
	GCD ENDP
	
	MAIN PROC
		MOV EAX,0
		MOV EBX,0
		CALL READINT
		MOV NUM1,EAX
		CALL READINT
		MOV NUM2,EAX
		CALL GCD
		CALL CRLF
		MOV EDX,OFFSET S
		CALL WRITESTRING
		CALL WRITEDEC
		CALL CRLF
	EXIT
	MAIN ENDP
END MAIN
