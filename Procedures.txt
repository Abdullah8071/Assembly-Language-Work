TITLE PROCEDURES


INCLUDE IRVINE32.INC

.DATA
S1 BYTE "SUM : ",0
X BYTE ?
Y BYTE ?

.CODE
	SUM PROC
		ADD EAX,EBX
		CALL WRITEDEC
		CALL CRLF
		RET
	SUM ENDP

	START PROC
		MOV EAX,0
		MOV EBX,0
		CALL READINT
		MOV EBX,EAX
		CALL READINT
		CALL DUMPREGS
		CALL CRLF
		MOV EDX,OFFSET S1
		CALL WRITESTRING
		CALL SUM
	START ENDP

	EXIT
END START
