TITLE BUBBLE SORT

INCLUDE IRVINE32.INC 

.DATA 
MYARRAY BYTE 15,10,25,20,30,50,54,52,51,53,57,59,60,62,63

.CODE 
	MAIN PROC
	MOV EAX,0
	MOV EBX,0
	MOV EDX,0 
	MOV ESI,0
	MOV EDI,1 
	MOV ECX,13
	L1: 
		MOV EAX,ECX 
		MOV ECX,14
		L2: 
			MOV BL,[MYARRAY+EDI] 
			CMP [MYARRAY+ESI],BL  
			JB NEXT 
			MOV BL,[MYARRAY+ESI] 
			MOV DL,[MYARRAY+EDI] 
			MOV [MYARRAY+ESI],DL 
			MOV [MYARRAY+EDI],BL
			NEXT: CALL FUNCTION   
		LOOP L2
		INC ESI 
		MOV EDI,ESI 
		INC EDI 
		MOV ECX,EAX 
	LOOP L1 
	MOV ECX,14
	MOV EAX,0  
	MOV ESI,0
	L3: 
		MOV AL,[MYARRAY+ESI]
		CALL WRITEDEC  
		CALL CRLF
		INC ESI 
		LOOP L3 
	FUNCTION PROC 
		CMP EDI,14 
		JE NEXT1 
		JNE NEXT2 

		NEXT1:RET 
		NEXT2:INC EDI 
		RET 
	FUNCTION ENDP 
	EXIT   
	MAIN ENDP 
END MAIN
