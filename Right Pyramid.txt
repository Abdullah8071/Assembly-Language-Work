TITLE RIGHT PYRAMID 


INCLUDE IRVINE32.INC 

.DATA 
LINES DWORD ?
SCALE DWORD 1
TEMP DWORD ?

.CODE 
	MAIN PROC
	MOV EAX,0
	CALL READINT
	MOV LINES,EAX
	MOV EBX,SCALE
	MOV ECX,LINES
	L1:
		MOV TEMP,ECX
		MOV ECX,LINES
		L2:			
			MOV AL,' '
			CALL WRITECHAR
		LOOP L2
		MOV ECX,SCALE
		L3:
			MOV AL,'*'
			CALL WRITECHAR
		LOOP L3
		INC SCALE
		DEC LINES
		CALL CRLF
		MOV ECX,TEMP
	LOOP L1

	EXIT   
	MAIN ENDP 
END MAIN
